456     /**
457      * Do the AspectJ compiling.
458      *
459      * @throws MojoExecutionException
460      */
461     @SuppressWarnings("unchecked")
462     public void execute() throws MojoExecutionException {
463 
464         if (isSkip()) {
465             if (getLog().isInfoEnabled()) {
466                 getLog().info("Skipping execution because of 'skip' option");
467             }
468             return;
469         }
470 
471         ArtifactHandler artifactHandler = project.getArtifact().getArtifactHandler();
472         if (!"java".equalsIgnoreCase(artifactHandler.getLanguage())) {
473             getLog().warn("Not executing aspectJ compiler as the project is not a Java classpath-capable package");
474             return;
475         }
476 
477         // MASPECT-110:
478         //
479         // Only add the aspectSourcePathDir and testAspectSourcePathDir to their respective
480         // compileSourceRoots if they actually exist and are directories... to avoid crashing
481         // downstream plugins requiring/assuming that all entries within the compileSourceRoots
482         // and testCompileSourceRoots are directories.
483         //
484         final File aspectSourcePathDir = FileUtils.resolveFile(basedir, aspectDirectory);
485         final File testAspectSourcePathDir = FileUtils.resolveFile(basedir, testAspectDirectory);
486 
487         final String aspectSourcePath = aspectSourcePathDir.getAbsolutePath();
488         final String testAspectSourcePath = testAspectSourcePathDir.getAbsolutePath();
489 
490         if (aspectSourcePathDir.exists() && aspectSourcePathDir.isDirectory()
491                 && !project.getCompileSourceRoots().contains(aspectSourcePath)) {
492             getLog().debug("Adding existing aspectSourcePathDir [" + aspectSourcePath + "] to compileSourceRoots.");
493             project.getCompileSourceRoots().add(aspectSourcePath);
494         } else {
495             getLog().debug("Not adding non-existent or already added aspectSourcePathDir [" + aspectSourcePath
496                     + "] to compileSourceRoots.");
497         }
498 
499         if (testAspectSourcePathDir.exists() && testAspectSourcePathDir.isDirectory()
500                 && !project.getTestCompileSourceRoots().contains(testAspectSourcePath)) {
501             getLog().debug(
502                     "Adding existing testAspectSourcePathDir [" + testAspectSourcePath + "] to testCompileSourceRoots.");
503             project.getTestCompileSourceRoots().add(testAspectSourcePath);
504         } else {
505             getLog().debug("Not adding non-existent or already added testAspectSourcePathDir [" + testAspectSourcePath
506                     + "] to testCompileSourceRoots.");
507         }
508 
509         assembleArguments();
510 
511         if (!forceAjcCompile && !hasSourcesToCompile()) {
512             getLog().warn("No sources found skipping aspectJ compile");
513             return;
514         }
515 
516         if (!forceAjcCompile && !isBuildNeeded()) {
517             getLog().info("No modifications found skipping aspectJ compile");
518             return;
519         }
520 
521         if (getLog().isDebugEnabled()) {
522             StringBuilder command = new StringBuilder("Running : ajc");
523 
524             for (String arg : ajcOptions) {
525                 command.append(' ').append(arg);
526             }
527             getLog().debug(command);
528         }
529         try {
530             getLog().debug(
531                     "Compiling and weaving " + resolvedIncludes.size() + " sources to " + getOutputDirectory());
532             AjcHelper.writeBuildConfigToFile(ajcOptions, argumentFileName, getOutputDirectory());
533             getLog().debug(
534                     "Arguments file written : " + new File(getOutputDirectory(), argumentFileName).getAbsolutePath());
535         } catch (IOException e) {
536             throw new MojoExecutionException("Could not write arguments file to the target area", e);
537         }
538 
539         final Main ajcMain = new Main();
540         MavenMessageHandler mavenMessageHandler = new MavenMessageHandler(getLog());
541         ajcMain.setHolder(mavenMessageHandler);
542 
543         synchronized (BIG_ASPECTJ_LOCK) {
544             ajcMain.runMain(ajcOptions.toArray(new String[ajcOptions.size()]), false);
545         }
546 
547         IMessage[] errors = mavenMessageHandler.getMessages(IMessage.ERROR, true);
548         if (!proceedOnError && errors.length > 0) {
549             throw CompilationFailedException.create(errors);
550         }
551     }

553     /**
554      * Assembles a complete ajc compiler arguments list.
555      *
556      * @throws MojoExecutionException error in configuration
557      */
558     protected void assembleArguments()
559             throws MojoExecutionException {
560         if (XhasMember) {
561             ajcOptions.add("-XhasMember");
562         }
563 
564         // Add classpath
565         ajcOptions.add("-classpath");
566         ajcOptions.add(AjcHelper.createClassPath(project, null, getClasspathDirectories()));
567 
568         // Add boot classpath
569         if (null != bootclasspath) {
570             ajcOptions.add("-bootclasspath");
571             ajcOptions.add(bootclasspath);
572         }
573 
574         if (null != Xjoinpoints) {
575             ajcOptions.add("-Xjoinpoints:" + Xjoinpoints);
576         }
577 
578         // Add warn option
579         if (null != warn) {
580             ajcOptions.add("-warn:" + warn);
581         }
582 
583         if (null != proc) {
584             ajcOptions.add("-proc:" + proc);
585         }
586 
587         if (Xset != null && !Xset.isEmpty()) {
588             StringBuilder sb = new StringBuilder("-Xset:");
589             for (Map.Entry<String, String> param : Xset.entrySet()) {
590                 sb.append(param.getKey());
591                 sb.append("=");
592                 sb.append(param.getValue());
593                 sb.append(',');
594             }
595             ajcOptions.add(sb.substring(0, sb.length() - 1));
596         }
597 
598         // Add artifacts or directories to weave
599         String joinedWeaveDirectories = null;
600         if (weaveDirectories != null) {
601             joinedWeaveDirectories = StringUtils.join(weaveDirectories, File.pathSeparator);
602         }
603         addModulesArgument("-inpath", ajcOptions, weaveDependencies, joinedWeaveDirectories,
604                 "dependencies and/or directories to weave");
605 
606         // Add library artifacts
607         addModulesArgument("-aspectpath", ajcOptions, aspectLibraries, getAdditionalAspectPaths(),
608                 "an aspect library");
609 
610         // Add xmlConfigured option and argument
611         if (null != xmlConfigured) {
612             ajcOptions.add("-xmlConfigured");
613             ajcOptions.add(xmlConfigured.getAbsolutePath());
614         }
615 
616         // add target dir argument
617         ajcOptions.add("-d");
618         ajcOptions.add(getOutputDirectory().getAbsolutePath());
619 
620         // Add all the files to be included in the build,
621         if (null != ajdtBuildDefFile) {
622             resolvedIncludes = AjcHelper.getBuildFilesForAjdtFile(ajdtBuildDefFile, basedir);
623         } else {
624             resolvedIncludes = getIncludedSources();
625         }
626         ajcOptions.addAll(resolvedIncludes);
627     }
628 
629     protected Set<String> getIncludedSources()
630             throws MojoExecutionException {
631         Set<String> result = new HashSet<String>();
632         if (getJavaSources() == null) {
633             result = AjcHelper.getBuildFilesForSourceDirs(getSourceDirectories(), this.includes, this.excludes);
634         } else {
635             for (int scannerIndex = 0; scannerIndex < getJavaSources().length; scannerIndex++) {
636                 Scanner scanner = getJavaSources()[scannerIndex];
637                 if (scanner.getBasedir() == null) {
638                     getLog().info("Source without basedir, skipping it.");
639                 } else {
640                     scanner.scan();
641                     for (int fileIndex = 0; fileIndex < scanner.getIncludedFiles().length; fileIndex++) {
642                         result.add(FileUtils.resolveFile(scanner.getBasedir(),
643                                 scanner.getIncludedFiles()[fileIndex]).getAbsolutePath());
644                     }
645                 }
646             }
647         }
648         return result;
649     }

651     /**
652      * Finds all artifacts in the weavemodule property, and adds them to the ajc options.
653      *
654      * @param argument
655      * @param arguments
656      * @param modules
657      * @param aditionalpath
658      * @param role
659      * @throws MojoExecutionException
660      */
661     private void addModulesArgument(final String argument, final List<String> arguments, final Module[] modules,
662                                     final String aditionalpath, final String role)
663             throws MojoExecutionException {
664         StringBuilder buf = new StringBuilder();
665 
666         if (null != aditionalpath) {
667             arguments.add(argument);
668             buf.append(aditionalpath);
669         }
670         if (modules != null && modules.length > 0) {
671             if (!arguments.contains(argument)) {
672                 arguments.add(argument);
673             }
674 
675             for (int i = 0; i < modules.length; ++i) {
676                 Module module = modules[i];
677                 // String key = ArtifactUtils.versionlessKey( module.getGroupId(), module.getArtifactId() );
678                 // Artifact artifact = (Artifact) project.getArtifactMap().get( key );
679                 Artifact artifact = null;
680                 @SuppressWarnings("unchecked") Set<Artifact> allArtifacts = project.getArtifacts();
681                 for (Artifact art : allArtifacts) {
682                     if (art.getGroupId().equals(module.getGroupId()) && art.getArtifactId().equals(
683                             module.getArtifactId()) && StringUtils.defaultString(module.getClassifier()).equals(
684                             StringUtils.defaultString(art.getClassifier())) && StringUtils.defaultString(
685                             module.getType(), "jar").equals(StringUtils.defaultString(art.getType()))) {
686                         artifact = art;
687                         break;
688                     }
689                 }
690                 if (artifact == null) {
691                     throw new MojoExecutionException(
692                             "The artifact " + module.toString() + " referenced in aspectj plugin as " + role
693                                     + ", is not found the project dependencies");
694                 }
695                 if (buf.length() != 0) {
696                     buf.append(File.pathSeparatorChar);
697                 }
698                 buf.append(artifact.getFile().getPath());
699             }
700         }
701         if (buf.length() > 0) {
702             String pathString = buf.toString();
703             arguments.add(pathString);
704             getLog().debug("Adding " + argument + ": " + pathString);
705         }
706     }

708     /**
709      * Checks modifications that would make us need a build
710      *
711      * @return <code>true</code> if build is needed, otherwise <code>false</code>
712      * @throws MojoExecutionException
713      */
714     protected boolean isBuildNeeded()
715             throws MojoExecutionException {
716         File outDir = getOutputDirectory();
717         return hasNoPreviousBuild(outDir) || hasArgumentsChanged(outDir) ||
718                 hasSourcesChanged(outDir) || hasNonWeavedClassesChanged(outDir);
719 
720     }
721 
722     private boolean hasNoPreviousBuild(File outDir) {
723         return !FileUtils.resolveFile(outDir, argumentFileName).exists();
724     }
725 
726     private boolean hasArgumentsChanged(File outDir)
727             throws MojoExecutionException {
728         try {
729             return (!ajcOptions.equals(AjcHelper.readBuildConfigFile(argumentFileName, outDir)));
730         } catch (IOException e) {
731             throw new MojoExecutionException("Error during reading of previous argumentsfile ");
732         }
733     }
734 
735     /**
736      * Not entirely safe, assembleArguments() must be run
737      */
738     private boolean hasSourcesToCompile() {
739         return resolvedIncludes.size() > 0;
740     }
741 
742     private boolean hasSourcesChanged(File outDir) {
743         long lastBuild = new File(outDir, argumentFileName).lastModified();
744         for (String source : resolvedIncludes) {
745             File sourceFile = new File(source);
746             long sourceModified = sourceFile.lastModified();
747             if (sourceModified >= lastBuild) {
748                 return true;
749             }
750 
751         }
752         return false;
753     }
754 
755     private boolean hasNonWeavedClassesChanged(File outDir)
756             throws MojoExecutionException {
757         if (weaveDirectories != null && weaveDirectories.length > 0) {
758             Set<String> weaveSources = AjcHelper.getWeaveSourceFiles(weaveDirectories);
759             long lastBuild = new File(outDir, argumentFileName).lastModified();
760             for (String source : weaveSources) {
761                 File sourceFile = new File(source);
762                 long sourceModified = sourceFile.lastModified();
763                 if (sourceModified >= lastBuild) {
764                     return true;
765                 }
766 
767             }
768         }
769         return false;
770     }

890     public void setXlintfile(File xlintfile) {
891         try {
892             final String prefix = "Xlintfile parameter invalid: ";
893             final String path = xlintfile.getCanonicalPath();
894             if (!xlintfile.exists()) {
895                 getLog().warn(prefix + " file [" + path + "] does not exist");
896             } else if (xlintfile.isDirectory()) {
897                 getLog().warn(prefix + " given path [" + path + "] is a directory.");
898             } else if (!path.trim().toLowerCase().endsWith(".properties")) {
899                 getLog().warn(prefix + " must be a .properties file");
900             } else {
901                 ajcOptions.add("-Xlintfile");
902                 ajcOptions.add(path);
903             }
904         } catch (IOException e) {
905             getLog().error("IOException while setting Xlintfile option", e);
906         }
907     }
